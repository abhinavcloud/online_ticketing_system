DO $$
BEGIN
  IF NOT EXISTS (SELECT 1 FROM pg_type WHERE typname = 'event_status') THEN
    CREATE TYPE event_status AS ENUM ('DRAFT', 'ON_SALE', 'SOLD_OUT', 'CANCELLED');
  END IF;

  IF NOT EXISTS (SELECT 1 FROM pg_type WHERE typname = 'seat_status') THEN
    CREATE TYPE seat_status AS ENUM ('AVAILABLE', 'BOOKED');
  END IF;

  IF NOT EXISTS (SELECT 1 FROM pg_type WHERE typname = 'reservation_status') THEN
    CREATE TYPE reservation_status AS ENUM ('HOLD', 'CONFIRMED', 'CANCELLED', 'EXPIRED', 'FAILED');
  END IF;
END $$;

-- ================================
-- TABLES
-- ================================
CREATE TABLE IF NOT EXISTS locations (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name TEXT UNIQUE
);

CREATE TABLE IF NOT EXISTS venues (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  location_id UUID REFERENCES locations(id),
  name TEXT,
  address TEXT
);

CREATE TABLE IF NOT EXISTS performers (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name TEXT UNIQUE
);

CREATE TABLE IF NOT EXISTS events (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name TEXT,
  venue_id UUID REFERENCES venues(id),
  event_date TIMESTAMPTZ,
  status event_status DEFAULT 'ON_SALE'
);

CREATE TABLE IF NOT EXISTS event_performers (
  event_id UUID REFERENCES events(id),
  performer_id UUID REFERENCES performers(id),
  PRIMARY KEY(event_id, performer_id)
);

CREATE TABLE IF NOT EXISTS event_categories (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  event_id UUID REFERENCES events(id),
  name TEXT,
  price BIGINT,
  currency TEXT,
  UNIQUE(event_id, name)
);

CREATE TABLE IF NOT EXISTS seats (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  event_id UUID REFERENCES events(id),
  category_id UUID REFERENCES event_categories(id),
  seat_label TEXT,
  status seat_status DEFAULT 'AVAILABLE',
  UNIQUE(event_id, seat_label)
);

CREATE TABLE IF NOT EXISTS reservations (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id TEXT,
  event_id UUID,
  category_id UUID,
  status reservation_status DEFAULT 'HOLD'
);

CREATE TABLE IF NOT EXISTS tickets (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id TEXT,
  reservation_id UUID,
  event_id UUID,
  category_id UUID,
  seat_label TEXT,
  total_amount BIGINT
);

-- ================================
-- CREATE EVENTS + CATEGORIES + SEATS
-- ================================
DO $$
DECLARE
  v_event_id UUID;
  v_venue_id UUID;
  v_cat_vip UUID;
  v_cat_gen UUID;
  i INT;
BEGIN

  FOR i IN 1..10 LOOP

    -- pick venue
    SELECT id INTO v_venue_id
    FROM venues
    LIMIT 1 OFFSET (i % (SELECT COUNT(*) FROM venues));

    -- create event
    INSERT INTO events(name, venue_id, event_date)
    VALUES ('Event ' || i, v_venue_id, now() + (i || ' days')::interval)
    RETURNING id INTO v_event_id;

    -- assign performer
    INSERT INTO event_performers(event_id, performer_id)
    SELECT v_event_id, id FROM performers LIMIT 1;

    -- categories
    INSERT INTO event_categories(event_id, name, price, currency)
    VALUES (v_event_id,'VIP',3000,'INR')
    RETURNING id INTO v_cat_vip;

    INSERT INTO event_categories(event_id, name, price, currency)
    VALUES (v_event_id,'GENERAL',500,'INR')
    RETURNING id INTO v_cat_gen;

    -- 1000 seats per category
    INSERT INTO seats(event_id, category_id, seat_label)
    SELECT v_event_id, v_cat_vip, 'VIP-' || gs
    FROM generate_series(1,1000) gs;

    INSERT INTO seats(event_id, category_id, seat_label)
    SELECT v_event_id, v_cat_gen, 'GEN-' || gs
    FROM generate_series(1,1000) gs;

  END LOOP;
END $$;
